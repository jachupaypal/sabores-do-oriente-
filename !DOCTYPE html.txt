function add(name,price){

function render(){
let list=document.getElementById('cart-items');
list.innerHTML='';
let total=0;
cart.forEach(item=>{
let li=document.createElement('li');
li.textContent=item.name+" - R$"+item.price;

 total+=item.price;
});

}

}
